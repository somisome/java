bind와 같은 경우

$('.bb').on('click',function(){
    $('<button/>',{
        'class':'bb',
        'text':'버튼'
        }).appendTo('#d2');
})

delegate와 같은 경우

$('div').on('click','.bb',function(){
    $('<button/>',{
        'class':'bb',
        'text':'버튼'
        }).appendTo('#d2');
})

bind, delegate 이벤트 해제

$('#but').bind('click',function(){
    $('div').off('click');
})
