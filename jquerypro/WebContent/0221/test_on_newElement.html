bind의 의미 : 기존의 h1에만 이벤트 핸들러가 실행
$('h1').on('click',function(){
$(this).css('background','orange');
$('<h1>새로운 엘리먼트</h1>').appendTo('body');
})


delegate의 의미 : 기존의 h1와 새로운 엘리먼트에도 이벤트 핸들러 실행
$('body').on('click','h1',function(){
$(this).css({'background':'pink','color':'tomato'});
$('<h1>새로운 엘리먼트</h1>').appendTo('body');
})
