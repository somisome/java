상태필터 checkbox:checked

$('input[type=checkbox]')
   모든 체크박스를 선택하는 것
$(':checkbox')
   역시 모든 체크박스를 선택하는 것
$(':checkbox :checked')
   사이에 띄어쓰기하면 checkbox중 자식만 선택한다는 뜻
$(':checkbox:checked')
   모든 checkbox선택하고 checked필터 이용

for문과 함께 사용하는 예

$(':checkbox:checked').each(function(i){
     if(i>0) result +=", ";
     result+=$(this).val();
})
