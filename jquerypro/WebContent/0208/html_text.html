.size()와 .length()

.text()와 .html()

.text(): 태그안의 모든 문자

.html(): 모든 태그 포함

alert($("body").html());
body 태그 안의 모든 태그를 포함한 문장
alert($("body").text());
body 태그안의 모든 문자만
alert($("p").text());

alert($("p").html());
p가 여러개인경우 첫번째 인거만 태그를 포함한 문장
