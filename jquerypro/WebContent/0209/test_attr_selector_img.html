type이 button인 input요소의 배경색 변경 $(
$('input[type=button]').css('background','lightblue');

href가 mailto 로 시작하는 a 요소의 배경색 변경 ^=
$('a[href^=mailto]').css('background','pink');

id 가 img1인것만 제외한(아닌) img요소의 테두리 변경 !=
$('img[id!=img1]').css('border','solid 3px tomato');

src가 png로 끝나는 img 요소의 테두리 변경 $=
$('img[src$=png]').css('border','solid 5px pink');

type이 text인 input 요소의 테두리
$('input[type=text]').css('border','solid 3px black');

title 이 named 단어(공백구분)를 포함한 img 요소 의 테두리 ~=
$('img[title~=named]').css('border','solid 3px sienna');

href가 power2c를 포함하는 모든 a 요소 의 밑줄 없음 *=
$('a[href*=power2c]').css('text-decoration','none');

id가 litte 이거나 little- 로 시작하는 img요소 의 테두리 |=
$('img[id|=little]').css('border','dotted 4px forestgreen');

alt 속성을 갖는 이미지 요소 의 테두리- 2px 단선 레드
$('img[alt]').css('border','solid 2px red');
